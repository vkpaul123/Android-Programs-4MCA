    @SuppressWarnings("StatementWithEmptyBody") 
    @Override 
    public boolean onNavigationItemSelected(MenuItem item) { 
        // Handle navigation view item clicks here. 
 
        FragmentManager fragmentManager = getSupportFragmentManager(); 
        FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction(); 
 
        int id = item.getItemId(); 
 
        if (id == R.id.createRecordItem) { 
            CreateRecordFragment createRecordFragment = new CreateRecordFragment(); 
 
            fragmentTransaction.replace(R.id.frag_container, createRecordFragment); 
            fragmentTransaction.commit(); 
 
            Toast.makeText(getApplicationContext(), "Create Record", Toast.LENGTH_SHORT).show(); 
        } else if (id == R.id.retrieveRecordItem) { 
            RetrieveRecordFragment retrieveRecordFragment = new RetrieveRecordFragment(); 
 
            fragmentTransaction.replace(R.id.frag_container, retrieveRecordFragment); 
            fragmentTransaction.commit(); 
 
            Toast.makeText(getApplicationContext(), "Retrieve Record", Toast.LENGTH_SHORT).show(); 
        } else if (id == R.id.updateRecordItem) { 
            UpdateRecordFragment updateRecordFragment = new UpdateRecordFragment(); 
 
            fragmentTransaction.replace(R.id.frag_container, updateRecordFragment); 
            fragmentTransaction.commit(); 
 
            Toast.makeText(getApplicationContext(), "Update Record", Toast.LENGTH_SHORT).show(); 
        } else if (id == R.id.deleteRecordItem) { 
            DeleteRecordFragment deleteRecordFragment = new DeleteRecordFragment(); 
 
            fragmentTransaction.replace(R.id.frag_container, deleteRecordFragment); 
            fragmentTransaction.commit(); 
 
            Toast.makeText(getApplicationContext(), "Delete Record", Toast.LENGTH_SHORT).show(); 
        } 
 
        DrawerLayout drawer = (DrawerLayout) findViewById(R.id.drawer_layout); 
        drawer.closeDrawer(GravityCompat.START); 
        return true; 
    }
